\section{Integrais}

Integrais em $\mathbb{R}^2$, com $\mathbb{D} \in \mathbb{R}^2$:
{\Large
    \begin{equation*}
        \int\int_{\mathbb{D}}f(x,y)\,dx\,dy
    \end{equation*}
}

\subsection{Tipos de domínio:}
\begin{itemize}
    \item \textbf{Tipo 1}:
        \subitem $\tcboxmath[colframe=red]{a<x<b} \qquad \wedge \qquad  \tcboxmath[colframe=blue]{f_1(x)<y<f_2(x)}$
        \subitem \begin{figure}[ht]
            \centering

                \begin{axis}[
                    axis lines=middle,

                ]
                    % Upper curve
                    \addplot[name path=upper1, blue, thick, domain=-2:1]
                        {e^x};
                    
                    % Lower curve

\subsection{Cálculo do limite}

Seja $\mathbb{D}$ um domínio de \textbf{tipo 1}:
\begin{boxedeq}[0.7]
    \int\int_{\mathbb{D}}f(x,y)\,dx\,dy \quad = \quad \int_{a}^{b}\left(\int_{f_1(x)}^{f_2(x)} f(x,y)\,dy\right)\,dx
\end{boxedeq}

Seja $\mathbb{D}$ um domínio de \textbf{tipo 2}:
\begin{boxedeq}[0.7]
    \int\int_{\mathbb{D}}f(x,y)\,dx\,dy \quad = \quad \int_{c}^{d}\left(\int_{g_1(x)}^{g_2(x)} f(x,y)\,dx\right)\,dy
\end{boxedeq}

Se o domínio for de tipo 1 e tipo 2, os dois integrais \textbf{são iguais}.
